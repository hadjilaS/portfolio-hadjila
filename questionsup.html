void triRecursive(int t[3][3], int size, int i, int j) {
  if (i == size) {
    return;
  }
  if (j < size - 1) {
    if (t[i][j] > t[i][j + 1]) {
      int tmp = t[i][j];
      t[i][j] = t[i][j + 1];
      t[i][j + 1] = tmp;
    }
    triRecursive(t, size, i, j + 1);
  } else {
    triRecursive(t, size, i + 1, 0);
  }
}
