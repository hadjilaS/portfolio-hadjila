2-La fonction somme pour calculer la somme des éléments du tableau.

3-La fonction recherche pour trouver une valeur spécifique à l'aide d'un pointeur.

Remarque: les pointeurs dans la fonction recherche permettent de renvoyer les indices de ligne et de colonne où la valeur spécifique a été trouvée, en modifiant directement les valeurs de ces indices à l'extérieur de la fonction.

        int remplissage(int tab[50][50], int n, int m, int i, int j) {

        //J'AI UTILISER LE "n" et le "m" pour definir l'odre du tableau et indices i et j pour chaque valeur :
        int somme(int tab[50][50], int n, int m, int i, int j) {
            if (i == n) {
                return 0;
            } else if (j == m) {
                return somme(tab, n, m, i + 1, 0);
            } else {
                return tab[i][j] + somme(tab, n, m, i, j + 1);
            }
        }
        //LA J'AI AJOUTER "ligne et colonnes "  qui pointe sur les adresse des indice comme jai mentionner dans laffichage de chaque indice pour preciser la location de la valeur specifique 
        int recherche(int tab[50][50], int b, int n, int m, int i, int j, int* ligne, int* colonne) {
            if (i == n) {
                return 0;
            } else if (j == m) {
                return recherche(tab, b, n, m, i + 1, 0, ligne, colonne);
            } else if (tab[i][j] == b) {
                *ligne = i;
                *colonne = j;
                return 1;
            } else {
                return recherche(tab, b, n, m, i, j + 1, ligne, colonne);
            }
        }
